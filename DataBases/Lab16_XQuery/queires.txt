Запит 5: Договори з кількома застрахованими особами
for $agreement in //agreement
let $insured_count := count($agreement//insured)
where $insured_count > 1
order by $insured_count descending
return 
  <multi_person_contract>
    <agreement_id>{data($agreement/@id)}</agreement_id>
    <status>{data($agreement/@status)}</status>
    <insured_persons_count>{$insured_count}</insured_persons_count>
    <total_amount>{data($agreement/total_insurance_amount)}</total_amount>
    <persons>
      {for $insured at $pos in $agreement//insured
       return 
         <person number="{$pos}">
           <name>{data($insured/person/pers/first_name)} {data($insured/person/pers/last_name)}</name>
           <gender>{data($insured/gender)}</gender>
           <phone>{data($insured/phone_number)}</phone>
         </person>}
    </persons>
  </multi_person_contract>

Запит 6: Статистика по валютах з умовою
for $currency in distinct-values(//prop/@currency)
let $properties := //prop[@currency=$currency]
let $total := sum($properties/value)
let $max_value := max($properties/value)
let $min_value := min($properties/value)
where count($properties) > 0
order by $total descending
return 
  <currency_statistics>
    <currency>{$currency}</currency>
    <properties_count>{count($properties)}</properties_count>
    <total_value>{$total}</total_value>
    <average_value>{round($total div count($properties))}</average_value>
    <max_value>{$max_value}</max_value>
    <min_value>{$min_value}</min_value>
    <has_premium>
      {if (some $prop in $properties satisfies $prop/value > 5000000) 
       then 'Так' 
       else 'Ні'}
    </has_premium>
  </currency_statistics>

(: Запит 2: Вивести кількість укладених компанією договорів страхування 
   та на яку сукупно суму застраховано вказані в них об'єкти за минулий квартал :)

let $agreements := //agreement[
  starts-with(conclusion_date, '2024-07') or 
  starts-with(conclusion_date, '2024-08') or 
  starts-with(conclusion_date, '2024-09')
]

let $total_count := count($agreements)
let $total_amount := sum($agreements/total_insurance_amount)

return
  <quarterly_report>
    <period>
      <quarter>Q3 2024 (липень - вересень)</quarter>
      <year>2024</year>
    </period>
    <statistics>
      <total_agreements>{$total_count}</total_agreements>
      <total_insured_amount>{round($total_amount, 2)}</total_insured_amount>
      <average_amount_per_agreement>
        {if ($total_count > 0) then round($total_amount div $total_count, 2) else 0}
      </average_amount_per_agreement>
    </statistics>
    <agreements>
      {for $agreement in $agreements
       order by $agreement/conclusion_date
       return
         <agreement>
           <id>{data($agreement/@id)}</id>
           <conclusion_date>{data($agreement/conclusion_date)}</conclusion_date>
           <amount>{data($agreement/total_insurance_amount)}</amount>
           <status>{data($agreement/@status)}</status>
         </agreement>}
    </agreements>
  </quarterly_report>
